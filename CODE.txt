 #include<Servo.h>
Servo s2;
int trig=4;
int echo=3;
void setup() {
  s2.attach(9);
   pinMode(trig, OUTPUT);  
   pinMode(echo, INPUT);
}
void loop(){
 long distance=GetDistance();
 delay(100);
 if( distance<=100 ){
    s2.write(90);
    delay(1000);
  }
  else{ 
   s2.write(0);
 	delay(1000);
   }
    delay(250);
}
long GetDistance(){
   long  dur,cm,in;
  digitalWrite(trig,LOW);
  delayMicroseconds(2);
  digitalWrite(trig,HIGH);
  delayMicroseconds(10);
  digitalWrite(trig,LOW);
  dur=pulseIn(echo,HIGH);
  cm=(dur/2)/29.1;
  in=cm/2.54;
  return in;
}
